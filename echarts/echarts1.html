ECharts图表三维数据的展示

怎样去使用一个ECharts demo

app.title = '多 Y 轴示例';



var colors = ['#5793f3', '#d14a61', '#675bba']; //可改动 颜色<<<<<<<<<<<<<<<<<<<<<



option = {

    color: colors, //可改动 颜色 , color为一个新的option关键字<<<<<<<<<<<<<<<<<<<<<

    title:{

      text:'标题'  //可改动 标题<<<<<<<<<<<<<<<<<<<<<

    },

    tooltip: {

        trigger: 'axis',

        axisPointer: {

            type: 'cross'

        }

    },

    grid: {

        right: '20%'    //可改动 位置<<<<<<<<<<<<<<<<<<<<<

    },

    toolbox: {

        feature: {

            dataView: {show: true, readOnly: false},

            restore: {show: true},

            saveAsImage: {show: true} //可改动 下载<<<<<<<<<<<<<<<<<<<<<

        }

    },

    legend: {

        data:['蒸发量','降水量','平均温度']  //可改动 栏目<<<<<<<<<<<<<<<<<<<<<

    },

    xAxis: [

        {

            type: 'category',

            axisTick: {

                alignWithLabel: true

            },

            data: ['1月','2月','3月','4月','5月','6月','7月','8月','9月','10月','11月','12月']  //可改动 x轴数据(时间)<<<<<<<<<<<<<<<<<<<<<

        }

    ],

    yAxis: [  //可改动 y轴数据<<<<<<<<<<<<<<<<<<<<<

        {

            type: 'value',

            name: '蒸发量',  

            min: 0,

            max: 250,

            position: 'right',

            axisLine: {

                lineStyle: {

                    color: colors[0]

                }

            },

            axisLabel: {

                formatter: '{value} ml' //可改动 单位<<<<<<<<<<<<<<<<<<<<<

            }

        },

        {

            type: 'value',

            name: '降水量',

            min: 0,

            max: 250,

            position: 'right',

            offset: 80,

            axisLine: {

                lineStyle: {

                    color: colors[1]

                }

            },

            axisLabel: {

                formatter: '{value} ml'  //可改动 单位<<<<<<<<<<<<<<<<<<<<<

            }

        },

        {

            type: 'value',

            name: '温度',

            min: 0,

            max: 25,

            position: 'left',

            axisLine: {

                lineStyle: {

                    color: colors[2]

                }

            },

            axisLabel: {

                formatter: '{value} °C'  //可改动 单位<<<<<<<<<<<<<<<<<<<<<

            }

        }

    ],

    series: [   //可改动 填充的数据<<<<<<<<<<<<<<<<<<<<<

        {

            name:'蒸发量',

            type:'bar',

            data:[2.0, 4.9, 7.0, 23.2, 25.6, 76.7, 135.6, 162.2, 32.6, 20.0, 6.4, 3.3]

        },

        {

            name:'降水量',

            type:'bar',

            yAxisIndex: 1,

            data:[2.6, 5.9, 9.0, 26.4, 28.7, 70.7, 175.6, 182.2, 48.7, 18.8, 6.0, 2.3]

        },

        {

            name:'平均温度',

            type:'line',

            yAxisIndex: 2,

            data:[2.0, 2.2, 3.3, 4.5, 6.3, 10.2, 20.3, 23.4, 23.0, 16.5, 12.0, 6.2]

        }

    ]

};
